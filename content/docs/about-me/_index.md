I love that moment when an idea goes from being just a thought to something you can actually hold in your hands.

For as long as I can remember, I've been taking things apart to see how they work, then putting them back together (usually better than before). Making, breaking, fixing, and redesigning things is what I do best. Over time, this turned into building actual projects like off-road machines, hydrogen buggies, and tons of mechanical systems that started as rough sketches and became real working parts.

CAD is where my ideas come to life. It's where messy thoughts turn into actual designs with real measurements and mechanisms that work in the real world.

I love solving problems that don't have instructions, designing things that don't exist yet, and taking projects from "what if?" to "it actually works!" For me, engineering isn't just about formulas or software , it's about creating things that move, improve, sometimes fail, and finally do something that matters.

## Skills

**CAD & Design:** Solidworks, Siemens NX, Fusion 360 | **Fabrication:** 3D Printing, Laser Cutting, CNC | **Programming:** Python, ESP32, Arduino

// Project carousels state
var projectCarousels = {};

function initProjectCarousels() {
  var containers = document.querySelectorAll('.project-carousel-container');
  containers.forEach(function(container) {
    var id = container.id;
    if (id) {
      projectCarousels[id] = { currentSlide: 0, interval: null };
      showProjectSlide(id, 0);
      startProjectAutoSlide(id);
    }
  });
}

function showProjectSlide(carouselId, n) {
  var container = document.getElementById(carouselId);
  if (!container) return;
  
  var slides = container.querySelectorAll('.project-carousel-slide');
  var state = projectCarousels[carouselId];
  
  if (n >= slides.length) { state.currentSlide = 0; }
  if (n < 0) { state.currentSlide = slides.length - 1; }
  
  slides.forEach(function(slide) { slide.classList.remove('active'); });
  if (slides.length > 0) {
    slides[state.currentSlide].classList.add('active');
  }
}

function changeProjectSlide(carouselId, n) {
  var state = projectCarousels[carouselId];
  if (!state) return;
  
  state.currentSlide += n;
  showProjectSlide(carouselId, state.currentSlide);
  
  // Reset auto-slide
  clearInterval(state.interval);
  startProjectAutoSlide(carouselId);
}

function startProjectAutoSlide(carouselId) {
  var state = projectCarousels[carouselId];
  if (!state) return;
  
  state.interval = setInterval(function() {
    state.currentSlide++;
    showProjectSlide(carouselId, state.currentSlide);
  }, 3000);
}

// Initialize project carousels when DOM is ready
document.addEventListener('DOMContentLoaded', initProjectCarousels);
</script>
